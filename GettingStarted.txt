To view all these commands on Terminal, run:: egrep '^\s+\$' GettingStarted.txt | sed -e 's@\$@@'


Commands::

    # Generate your digitalocean API token: https://cloud.digitalocean.com/account/api/tokens
    $ export TF_VAR_do_token="type_your_token_here"
    
    # Prepare your working directory for other commands
    $ terraform init
    
    # Check whether the configuration is valid
    $ terraform validate
    
    # Show changes required by the current configuration
    $ terraform plan -out=.tfplan -var-file=fra1.tfvars
    
    # Create or update infrastructure
    $ terraform apply -var-file=fra1.tfvars
    
    # Destroy previously created infrastructure
    $ terraform destroy
    
    # Unset token
    $ unset TF_VAR_do_token
